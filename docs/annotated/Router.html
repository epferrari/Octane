	/**
	* Remove the root from a URL, using Application webRoot or localRoot, if defined, depending on context
	* webRoot is defined as `appConfig.webRoot` in **Octane.initialize**, else defaults to `location.origin + '/' + location.pathname`
	* localRoot is defined as `appConfig.localRoot` in **Octane.initialize**, else defaults to local filepath up to 'index.html'
	*
	* @private
	* @static
	* @method pruneRoot
	* @returns {string} the URL with root removed
	*/
	Router.defineProp('pruneRoot',function(url){

		var root,localRoot;
		if(localRouting){
			localRoot = _octane.localRoot ? new RegExp('^.*'+_octane.localRoot) : /^(.*index.html)/;
			return url.replace(localRoot,'');
		} else {
			root = _octane.webRoot || location.origin+'/'+location.pathname;
			return url.replace(root,'');
		}
	});

	/**
	* Apply the route logic to display correct Page and Application state when the URL changes, should not be called on its own, use `sendRoute` instead.
	*
	* @public
	* @static
	* @method route
	* @param {string} url A URL to match against defined routes
	*/
	Router.defineProp('executeRoute',function(url){

		var path = Router.pruneRoot(url);
		_.each(routes,function(route){
			if(route.re.test(path)) {
				var params = path.match(route.re);
				params.shift();
				route.fn.apply(Object.create(null),params);
			}
		});
	});


	/**
	* Direct page loading and exit animations
	*
	* @see OctanePage
	* @public
	* @static
	* @method loadPage
	* @param {string} page The name of the page to be loaded
	* @returns {Promise} a thenable that the load animation sequence for all requested pages has completed
	*/
	Router.defineProp('loadPage',function(page){
		return new Promise(function(resolve,reject){
			var P = _octane.pages[page];
			var pageIsCurrent = (P === Router.currentPage);

			Router.fire('routing:called');

			if(!P || pageIsCurrent) {
				Router.fire('routing:complete');
				return resolve();
			}

			if(pageAnimating || routingLocked){
				var toQueue = {
					page: 		page,
					resolver: resolve
				};
				if(!_.contains(queuedPages,toQueue)){
					queuedPages.unshift(toQueue);
				}
				return;
			}

			if( pageHistory[1] === P ){
				return P.frameWillLoad()
				.then(function(pg){
					Router.fire('routing:begin');
					pageAnimating = pg;
					return pg._queue();
				})
				.then(function(){
					var currentPage = Router.currentPage;
					if(currentPage){
						currentPage.elem.classList.add('frame-animating');
						return currentPage._exit();
					}
				})
				.then(function(currentPage){
					currentPage && currentPage.elem.classList.remove('frame-animating');
					return P._load();
				})
				.then(P.frameDidLoad)
				.then(function(){
					Router.fire('page:loaded');
					pageHistory.shift();
					pageAnimating = null;
					App.set({
						pageID: 		P.id,
						pageTitle: 	P.title
					});
				})
				.then(function(){
					if(queuedPages.length >0){
						var next = queuedPages.shift();
						return Router.loadPage(next.page).then(next.resolver);
					}
				})
				.then(function(){
					resolve();
					Router.fire('routing:complete');
				})
				.catch(function(err){
					Router.log(err);
				});

			} else {

				return P.frameWillLoad()
				.then(function(page){
					Router.fire('routing:begin');
					pageAnimating = page;
					if(Router.currentPage) return Router.currentPage._queue();
				})
				.then(function(){
					Router.fire('page:loading');
					return P._load();
				})
				.then(P.frameDidLoad)
				.then(function(page){
					Router.fire('page:loaded');
					pageHistory.unshift(page);
					pageAnimating = null;
					App.set({
						pageID: 		page.id,
						pageTitle: 	page.title
					});
					if(pageHistory[1]) pageHistory[1]._exit();
				})
				.then(function(){

					if(queuedPages.length >0){
						var next = queuedPages.shift();
						return Router.loadPage(next.page).then(next.resolver);
					}
				})
				.then(function(){
					resolve();
					Router.fire('routing:complete');
				})
				.catch(function(err){
					Router.log(err);
				});
			}
		});
	});
